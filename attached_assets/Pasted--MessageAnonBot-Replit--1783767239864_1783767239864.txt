Создать полную копию бота @MessageAnonBot на Replit с нуля прямо здесь невозможно, так как это требует настройки сервера и базы данных. Однако я дам вам готовый, рабочий код на Python, который вы сможете скопировать, вставить в Replit и запустить за 5 минут. Он реализует всю основную логику анонимного чата 1 на 1.

Полный код бота для Replit (Python + pyTelegramBotAPI)

Создайте новый Replit с типом Python, вставьте этот код в файл main.py и нажмите "Run":
import telebot
import threading
import time
import random
from flask import Flask, request
import os

# --- 1. Конфигурация ---
BOT_TOKEN = os.environ.get('BOT_TOKEN')  # Обязательно установите секрет!
if not BOT_TOKEN:
    raise Exception("❌ Установите BOT_TOKEN в Secrets Replit!")

bot = telebot.TeleBot(BOT_TOKEN)
app = Flask(__name__)

# --- 2. Хранилище состояний (в памяти) ---
# В реальном проекте используйте БД (SQLite/PostgreSQL)
waiting_users = []          # Очередь ищущих (user_id)
active_chats = {}           # {user1_id: user2_id, user2_id: user1_id}

# --- 3. Команды бота ---
@bot.message_handler(commands=['start', 'help'])
def send_welcome(message):
    bot.reply_to(message,
        "👋 *Анонимный чат 1 на 1*\n\n"
        "Ищите собеседника для общения или знакомств!\n"
        "Ваша личность и ID скрыты.\n\n"
        "🔍 /search — Начать поиск собеседника\n"
        "⏹ /stop — Выйти из чата и поиска\n"
        "❓ /help — Это сообщение",
        parse_mode="Markdown"
    )

@bot.message_handler(commands=['search'])
def search_partner(message):
    user_id = message.chat.id

    # Если уже в чате
    if user_id in active_chats:
        bot.reply_to(message, "⚠️ Вы уже в чате. Используйте /stop, чтобы выйти.")
        return

    # Если уже в очереди
    if user_id in waiting_users:
        bot.reply_to(message, "⏳ Вы уже в поиске. Ожидайте...")
        return

    # Добавляем в очередь
    waiting_users.append(user_id)
    bot.reply_to(message, "🔍 Ищем собеседника...")

    # Пытаемся найти пару
    try_match()

def try_match():
    """Ищет пару для первого в очереди"""
    if len(waiting_users) < 2:
        return

    user1 = waiting_users.pop(0)
    user2 = waiting_users.pop(0)

    # Создаем чат
    active_chats[user1] = user2
    active_chats[user2] = user1

    bot.send_message(user1, "🎉 *Собеседник найден!* Можете общаться.", parse_mode="Markdown")
    bot.send_message(user2, "🎉 *Собеседник найден!* Можете общаться.", parse_mode="Markdown")

@bot.message_handler(commands=['stop'])
def stop_chat(message):
    user_id = message.chat.id

    # Выход из чата
    if user_id in active_chats:
        partner_id = active_chats[user_id]
        del active_chats[user_id]
        del active_chats[partner_id]
        bot.send_message(user_id, "⏹ Вы вышли из чата.")
        bot.send_message(partner_id, "⏹ Собеседник покинул чат. Используйте /search для нового.")
        return

    # Выход из очереди
    if user_id in waiting_users:
        waiting_users.remove(user_id)
        bot.reply_to(message, "⏹ Поиск отменен.")
        return

    bot.reply_to(message, "⚠️ Вы не в чате и не в поиске.")

# --- 4. Обработка сообщений в чате ---
@bot.message_handler(func=lambda message: True)
def handle_message(message):
    user_id = message.chat.id

    # Проверяем, есть ли собеседник
    if user_id not in active_chats:
        bot.reply_to(message, "❌ Вы не в чате. Используйте /search для поиска.")
        return

    partner_id = active_chats[user_id]

    try:
        # Пересылаем сообщение собеседнику
        if message.text:
            bot.send_message(partner_id, message.text)
        elif message.photo:
            bot.send_photo(partner_id, message.photo[-1].file_id, caption=message.caption)
        elif message.sticker:
            bot.send_sticker(partner_id, message.sticker.file_id)
        elif message.voice:
            bot.send_voice(partner_id, message.voice.file_id)
        else:
            bot.reply_to(message, "⚠️ Я могу пересылать текст, фото, стикеры и голосовые.")
    except Exception as e:
        bot.reply_to(message, f"❌ Ошибка: {e}")
